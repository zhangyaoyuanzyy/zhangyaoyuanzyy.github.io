基本语法

  • 数据渲染：{{}}
    <p v-text="html1"></p>
    <input type="text" v-model="message"/> 属性渲染：v-bind
    <h1 v-bind:title="message">hello</h1>
    <a :href="url">hello</a>
  • 类名渲染：
    <div v-bind:class="active : isActive"></div>
    <div v-bind:class="isActive ? active : red"> </div> 样式渲染：
    <div v-bind:style="{width:width}"></div>
    <div v-bind:sytle="styleObj"></div>
    <div v-bind:style="[styleObj1,styleObj2]"></div>
  • 条件绑定
    <p v-if="seen">hahah</p>//有无内容
    <p v-show="seen">hah</p>//控制显隐
    循环绑定：
    <p v-for="list in lists">{{alist.text}}</p>
    <p v-for="(key value) in lists">{{key}}:{{value}}</p>
    <p v-for="(index key value) in lists">{{index}}:{{key}}:{{value}}</p>
  • 事件绑定：
    <a v-on:click="fun1">点击</a>
    <a @click="fun1">点击</a>
    事件修饰符：
    <a v-on:click.stop="doThis"></a>
    <form v-on:submit.prevent></form>
    <div v-on:click.capture="doThis">...</div>
  • 运算绑定：
    <p>{{reverseMess}}<p/>
